187 void CPhysicsEngine::applyPhysics(DPhysics* pObject)
188 {
189  sf::Time timer = pObject->m_sPhysics.gravityTimer.getElapsedTime();
190 
191  double elapsed = timer.asMilliseconds();
192 
193  // make game look nice
194  // remember 'elapsed' is in MS, this reduces that massive x var
195  double reductionFactor = 0.005;
196 
197  bool isJumping = pObject->m_sPhysics.isJumping;
198  bool isFalling = pObject->m_sPhysics.isFalling;
199 
200  if (isJumping)
201  {
202  pObject->m_sPhysics.velosity_y = 3.0f * (reductionFactor * elapsed) - 5;
203  }
204 
205  if (isFalling)
206  {
207  pObject->m_sPhysics.velosity_y = 3.0f * (reductionFactor * elapsed);
208  }

308 void CPhysicsEngine::applyPhysics(DPhysics* pObject)
309 {
310  sf::Time timer = pObject->m_sPhysics.gravityTimer.getElapsedTime();
311 
312  double elapsed = timer.asMilliseconds();
313 
314  // make game look nice
315  // remember 'elapsed' is in MS, this reduces that massive x var
316  double reductionFactor = 0.005;
317 
318  bool isJumping = pObject->m_sPhysics.isJumping;
319  bool isFalling = pObject->m_sPhysics.isFalling;
320 
321  if (isJumping)
322  {
323  pObject->m_sPhysics.velosity_y = 3.0f * (reductionFactor * elapsed) - 5;
324  }
325 
326  if (isFalling)
327  {
328  pObject->m_sPhysics.velosity_y = 3.0f * (reductionFactor * elapsed);
329  }
330 }
